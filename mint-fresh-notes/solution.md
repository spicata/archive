$$
\begin{align}
\frac{a}{x+a}+\frac{b}{x-b}&=\frac{a+b}{x+c} \\
\frac{a(x-b)+b(x+a)}{(x+a)(x-b)}&= \\
(x+c)(a(x-b)+b(x+a))&=(a+b)(x+a)(x-b) \\
(x+c)(ax-ab+bx+ab)&= \\
(x+c)(ax+bx)&= \\
x(x+c)(a+b)&= \\
x(x+c)&=(x+a)(x-b) \\
x^2+cx&=x^2+ax-bx-ab \\
x^2-cx-x^2-ax+bx&=-ab \\
-cx-ax+bx&= \\
x(-c-a+b)&=-ab \\
x&=\frac{-ab}{-c-a+b} \\
x&=\frac{ab}{a-b+c}
\end{align}
$$
